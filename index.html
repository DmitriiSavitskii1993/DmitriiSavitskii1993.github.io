from aiogram import Bot, Dispatcher
from openai import AsyncOpenAI

client = AsyncOpenAI()

@dp.message()
async def ai_reply(msg: Message):
    response = await client.chat.completions.create(
        model="gpt-4o",
        messages=[{
            "role": "user",
            "content": msg.text
        }]
    )
    await msg.answer(
        response.choices[0].message.content
    )
